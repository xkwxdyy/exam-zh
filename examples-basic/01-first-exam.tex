\section{解答题（共 10 分）}

\begin{problem}[points = 10]
  计算：$(x + 1)^2$

  \begin{solution}
    解：

    $(x + 1)^2$

    $= (x + 1)(x + 1)$

    $= x^2 + x + x + 1$

    $= x^2 + 2x + 1$

    答：$(x + 1)^2 = x^2 + 2x + 1$
  \end{solution}
\end{problem}
